void
fifo_page_replacement (char *queue_frames, int num)
{
  // TODO
  int *frames = (int *)malloc (num * sizeof (int));
  int queue_size = 0;
  int queue_index = 0;

  for (int i = 0; i < num; i++) {
    frames[i] = -1; // åˆå§‹åŒ–é¡µæ¡†
  }

  char *token = strtok (queue_frames, ",");
  while (token != NULL) {
    int page = atoi (token);
    printf("Access:%d,Frames: [", page);
    
    int found = 0;

    for (int i = 0; i < num; i ++) {
      if (frames[i] == page) {
        found = 1;
        break;
      }
    }

    if (!found) {
      if (queue_size < num) {
        frames[queue_size ++] = page;
      } else {
        frames[queue_index] = page;
        queue_index = (queue_index + 1) % num;
      }
    }

    for (int i = 0; i < num; i++) {
      printf("%d", frames[i]);
      if (i != num - 1) printf(",");
    }
    printf("]\n");

    token = strtok(NULL, ",");
  }

  free(frames);
}
