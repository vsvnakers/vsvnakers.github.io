int fifo_page_replacement(int pages[], int n, int frames) {
    int memory[frames];
    for (int i = 0; i < frames; i++) memory[i] = -1;

    int index = 0, pageFaults = 0;

    for (int i = 0; i < n; i++) {
        int found = 0;
        for (int j = 0; j < frames; j++) {
            if (memory[j] == pages[i]) {
                found = 1;
                break;
